01变量绑定与结构

变量绑定与解构

1.1 变量绑定

// 变量绑定： a 变量绑定1，拥有对1的所有权
let a： i32 = 1；
// 错误代码
a += 1;

1.2 可变变量

// 可变变量:
let mut a: i32 = 12;
// 
// 正确代码.如果没有mut 修饰则是错误的
a += 1;

1.3 作用域和遮蔽

// 变量绑定一个作用域scope, 它被限定在代码块中生存。另外允许变量遮蔽
fn main() {
    // 绑定是main 函数
    let long_lived = 1;

    // 这是一个代码块
    {
        // 绑定此代码块
        let short_lived = 2;

        // 此绑定*遮蔽*了外面的绑定
        let long_lived= 5_f32;       
    }
    // 代码块结束，short_lived 被释放
    println!("outer long: {}", long_lived);
}

1.4 变量冻结

// 被相同名称的不变地绑定时候，它会被冻结
fn main() {
    let mut _mutable_a = 7i32;

    {
        // 被不可变的 _mutable_a 遮蔽
        let _mutable_a = _mutable_a;

        // 在本作用域被冻结,会报错的
        _mutable_a = 50;
    }
    // 正常运行！ `_mutable_a` 在这个作用域没有冻结
    _mutable_a = 3;
}

1.5 变量解构

// 从复杂的结构解构初简单的变量
fn main() {
    let (a, mut b): (bool, bool) = (true, true);
    println!("a = {:?}, b = {:?}", a, b);

}
